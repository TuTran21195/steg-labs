```python
# step2_extract_bits.py
import sys
import numpy as np
import scipy.signal as signal
import scipy.io.wavfile as wav
import os

def bandpass_filter(sig, lowcut, highcut, fs, order=6):
    nyq = 0.5 * fs
    b, a = signal.butter(order, [lowcut/nyq, highcut/nyq], btype='band')
    return signal.filtfilt(b, a, sig)

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("C√°ch d√πng: python3 step2_extract_bits.py <input_wav> <hopping_pattern.txt>")
        sys.exit(1)

    input_wav = sys.argv[1]
    hopping_file = sys.argv[2]
    output_bits = "extracted_bits.txt"  # Output c·ªë ƒë·ªãnh

    if not os.path.exists(input_wav) or not os.path.exists(hopping_file):
        print(f"L·ªói: File kh√¥ng t·ªìn t·∫°i.")
        sys.exit(1)

    fs, noisy_signal = wav.read(input_wav)
    noisy_signal = noisy_signal.astype(np.float32) / 32768.0

    with open(hopping_file, 'r') as f:
        hopping_pattern = list(map(int, f.read().strip().split()))

    # C·∫•u h√¨nh
    bit_duration = 0.01
    base_freq = 2000
    delta = 100

    samples_per_bit = int(bit_duration * fs)
    num_bits = len(noisy_signal) // samples_per_bit # s·ªë bit c√≥ th·ªÉ c√≥ trong file √¢m thanh

    # D√πng hopping_pattern quay v√≤ng
    if len(hopping_pattern) < num_bits:
        times = (num_bits // len(hopping_pattern)) + 1
        hopping_pattern = (hopping_pattern * times)[:num_bits]

    extracted_bits = []

    for i in range(num_bits):
        segment = noisy_signal[i * samples_per_bit: (i + 1) * samples_per_bit]
        hop = hopping_pattern[i]

        f0 = base_freq + hop * delta
        f1 = base_freq + (hop + 1) * delta

        filtered_f0 = bandpass_filter(segment, f0 - 30, f0 + 30, fs)
        filtered_f1 = bandpass_filter(segment, f1 - 30, f1 + 30, fs)

        power_f0 = np.sum(filtered_f0 ** 2)
        power_f1 = np.sum(filtered_f1 ** 2)

        if power_f1 > power_f0:
            extracted_bits.append('1')
        else:
            extracted_bits.append('0')

    with open(output_bits, 'w') as f:
        for bit in extracted_bits:
            f.write(str(bit))

    print(f"ƒê√£ l∆∞u d√£y bit t·∫°i: {output_bits}")
```

```python
# step3_bits_to_message.py
import sys
import os

def bits_to_string(bits):
    chars = []
    for i in range(0, len(bits), 8):
        byte = bits[i:i+8]
        if len(byte) < 8:
            continue  # b·ªè qua n·∫øu thi·∫øu bit cu·ªëi
        char = chr(int(byte, 2))
        if char == '\0':   # G·∫∑p k√≠ t·ª± null th√¨ d·ª´ng l·∫°i
            break
        chars.append(char)
    return ''.join(chars)

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("C√°ch d√πng: python3 step3_bits_to_message.py <input_bits.txt>")
        sys.exit(1)

    input_bits = sys.argv[1]
    output_message = "message.txt"  # Output c·ªë ƒë·ªãnh

    if not os.path.exists(input_bits):
        print(f"L·ªói: File {input_bits} kh√¥ng t·ªìn t·∫°i.")
        sys.exit(1)

    with open(input_bits, 'r') as f:
        bits = f.read().strip()

    message = bits_to_string(bits)

    with open(output_message, 'w') as f:
        f.write(message)

    print(f"ƒê√£ l∆∞u th√¥ng ƒëi·ªáp t·∫°i: {output_message}")
    print(f"Th√¥ng ƒëi·ªáp tr√≠ch xu·∫•t ƒë∆∞·ª£c: {message}")
```

```python
# step4_calculate_ber.py
import sys
import os


if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("C√°ch d√πng: python3 step4_calculate_ber.py <extracted_bits.txt> <original_bits.txt>")
        sys.exit(1)

    extracted_bits = sys.argv[1] # file bit v·ª´a tr√≠ch xu·∫•t ƒë∆∞·ª£c
    original_bits = sys.argv[2]  # file bit g·ªëc (ng∆∞·ªùi nh·∫≠n kh√¥ng c√≥, nh∆∞ng ƒë·ªÉ ƒë√°nh gi√° ph∆∞∆°ng ph√°p th√¨ ta s·∫Ω d√πng n√≥)
    output_ber = "ber.txt"  # Ghi BER ra file

    if not os.path.exists(extracted_bits) or not os.path.exists(original_bits):
        print(f"L·ªói: M·ªôt trong c√°c file kh√¥ng t·ªìn t·∫°i.")
        sys.exit(1)

    with open(extracted_bits, 'r') as f:
        bits1 = f.read().strip()

    with open(original_bits, 'r') as f:
        bits2 = f.read().strip()

	# ======= T√≠nh BER ==========
    length = min(len(bits1), len(bits2))
    errors = sum(b1 != b2 for b1, b2 in zip(bits1[:length], bits2[:length]))
    ber = errors / length
    return ber

    with open(output_ber, 'w') as f:
        f.write(f"BER: {ber:.6f}")

	print(f"üìà BER (Bit Error Rate): {ber:.6f} ({errors} errors tr√™n {length} bits)")
    print(f"ƒê√£ l∆∞u BER t·∫°i: {output_ber}")
```
